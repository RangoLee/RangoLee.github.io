C++ 中new和new()的区别

结论：new()调用没有参数的构造函数，new调用默认构造函数或唯一构造函数。 C++在new时的初始化规律可能为：对于有构造函数的类，new和new()都调用构造函数进行初始化； 如果没有构造函数，则new只分配内存空间，不初始化；new()分配内心空间的同时初始化。

int main()
{
    int *pAA = new int[100];
    for (int i = 0; i < 100; ++i)
    {
    	pAA[i] = i+1;
    	// std::cout << pAA[i] << " ";
    }
    delete[] pAA;

    int *pBB = new int[100]; // 不加()会出现随机值   
    for (int i = 0; i < 100; ++i)
    {
    	std::cout << pBB[i] << " ";
    }
    std::cout << std::endl;
    delete[] pBB;
    
    int *pCC = new int[100](); // 加了()后都是0
    for (int i = 0; i < 100; ++i)
    {
    	std::cout << pCC[i] << " ";
    }
    std::cout << std::endl;

    delete[] pCC;
    return 0;
}

#include <iostream>

class A
{
public:
    A():n(10){};
    int n;
};
int main()
{
    A *pA = new A; 
    std::cout << pA->n << std::endl; // n可能会是随机值
    
    A *pA2 = new A();
    std::cout << pA2->n << std::endl; // n的值是0
    return 0;
}
